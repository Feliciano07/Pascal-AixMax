// funcion factorial y numeo minimo


program functionsandprocs;
function factorial(n: integer): integer;
begin
  if n>1 then
    factorial:=n*factorial(n-1)
  else
    factorial:=1;
end;

var num, f, m, n, r: integer;

begin
  //write('n = ');
  //readln(n);
  num:=7;
  f:=factorial(num);
  writeln(num,'! = ',f);
end.
